public async Task<IActionResult> ViewFile(int? id) {
    if (id == null)
        return NotFound();

    var file = await _context.Artifacts
        .Where(x => x.Id == id)
        .Select(x => x.File)
        .FirstOrDefaultAsync();

    return File(file, System.Net.Mime.MediaTypeNames.Application.Pdf);
}
